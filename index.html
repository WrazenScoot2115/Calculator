function calculator(operator, num1, num2) { if (operator === '+') { return num1 + num2; } else if (operator === '-') { return num1 - num2; } else if (operator === '*') { return num1 * num2; } else if (operator === '/') { return num1 / num2; } else { return 'Nieprawidłowy operator'; } } // Przykład użycia: let result = calculator('+', 5, 3); console.log(result); // 8
